\bconj{[GFR] \label{GFR} Every graph $G$ with $\alpha(G) \leq 2$ and $|V(G)| \geq 4t-1$ has a $t$-connected matching.}

\bconj{[GFRc] \label{GFRc} There is a constant $c$ so that every graph $G$ with $\alpha(G) \leq 2$ and $|V(G)| \geq ct$ has a $t$-connected matching.}


\bprop{Conjecture \ref{GFRc} holds for any class $\mathcal{P}$ of graphs such that $\overline{\mathcal{P}}$ is chromatically bounded.}
\begin{proof}
	Let $\mathcal{P}_2 = \{G \in \mathcal{P}: \alpha(G)\leq 2\}$ and suppose that every element of $\overline{\mathcal{P}_2}$ can be $d$ colored.  Then every element $G$ of $\mathcal{P}_2$ has a clique of size $|V(G)|/d$, and by THE LEMMA a $|V(G)|/d$-connected matching. 
\end{proof}
